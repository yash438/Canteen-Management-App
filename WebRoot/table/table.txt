//table

create table product (name varchar(20) primary key,quantity int not null,price float not null,category_id varchar(20) not null);

create table user (username varchar(20) primary key,name varchar(30),pass varchar(10),mobile varchar(10),address varchar(40));

create table orders(product_name varchar(30),product_price varchar(10),product_quantity varchar(5),total varchar(10),deliver varchar(20),name varchar(20),mobile varchar(10),uname varchar(20),mode varchar(10),order_id int);

create table payment (username varchar(20),mobile varchar(10),amount varchar(20),status varchar(10),order_id int);
